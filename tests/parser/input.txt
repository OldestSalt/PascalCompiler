program Maximum;
var max: integer;
i: integer;
ch: integer;
n: string;
begin
write('Введите количество чисел: ');
readln(n);
i:=1.0;
for i:=1 to n + 'string' do
begin
     write('Введите ',i,'-е число: ');
     readln(ch);
     if max<ch then max:=ch;
end;
writeln('Самое большое число: ', max);
end.
